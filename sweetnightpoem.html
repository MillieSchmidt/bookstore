הזדמנות שנייה

אֲנִי לֹא רוֹצֶה הִזְדַּמְּנוּת שְׁנִיָּה
בְּשׁוּם מָקוֹם בְּשׁוּם דָּבָר עִם אַף אַחַת
אֵין לִי זְמַן לַחְזֹר אָחוֹרָה וּלְתַקֵּן טָעֻיּוֹת
שֶׁלֹּא כֻּלָּן שֶׁלִּי בְּמֵילָא
מָה שֶׁהָיָה הוּא שֶׁיִּהְיֶה
אָז בֶּאֱמֶת שֶׁאֵין לִי טַעַם לְנַסּוֹת שׁוּב
יוֹתֵר מִדַּי אֲנָשִׁים מְכוּרִים לַעֲבַר שֶׁלָּהֶם
הַלְוַאי שֶׁהֵם הָיוּ מְכוּרִים בְּאוֹתָהּ צוּרָה
לִפְעִילוּת גּוּפָנִית וַאֲכִילָה נְכוֹנָה
וּלְחַסְכָנוּת בִּמְקוֹם לַאֲכִילַת גְּלִידָה
שֶׁאֲמוּרָה לְנַחֵם אוֹתָם עַל דְּבָרִים שֶׁהֵם לֹא יְכוֹלִים לַעֲשׂוֹת יוֹתֵר
יוֹתֵר מִדַּי אֲנָשִׁים רוֹצִים לַחְזֹר לְזֶה אוֹ לְזֹאתִי
אוֹ לְכָל מִינֵי מְקוֹמוֹת שֶׁהֵם בִּקְּרוּ בָּהֶם
מָה הַטַּעַם
הָיִיתִי שָׁם וְעָשִׂיתִי אֶת זֶה
וְאֵין חָדָשׁ תַּחַת הַשֶּׁמֶשׁ
אֲנִי לֹא רוֹצֶה לָרוּץ לְאָחוֹר
לִרְדֹּף אַחֲרֵי עוֹלָם שֶׁנֶּעֱלַם
אֲנִי מִסְתַּכֵּל כָּל הַזְּמַן עַל הָרַגְלַיִם שֶׁלִּי
צַעַד אַחֲרֵי צַעַד
לֹא אִכְפַּת לִי מָה יָבוֹא הָלְאָה
אֲבָל לְפָחוֹת זֶה מַשֶּׁהוּ שֶׁאֶפְשָׁר לְהִתְכּוֹנֵן אֵלָיו
לֹא לְפַתֵּחַ צִפִּיּוֹת
לִחְיוֹת יוֹם יוֹם
אֲבָל זֶה לֹא גָּרוּעַ כְּמוֹ לִחְיוֹת אֶתְמוֹל
וּלְחַפֵּשׂ עוֹד הִזְדַּמְּנוּת וְעוֹד אֶפְשָׁרֻיּוֹת שֶׁנִּסְגְּרוּ כְּבָר
פַּעַם חִפַּשְׂתִּי בְּחַיִּים שֶׁל אֲחֵרִים
אֶת כָּל מָה שֶׁלֹּא הָיָה בְּחַיַּי
זֶה הָיָה בְּסֵדֶר לִתְקוּפָה מְסֻיֶּמֶת אֲבָל לֹא יוֹתֵר מִזֶּה
כִּי זֶה לֹא אֲנִי
לָאֲנָשִׁים הָהֵם כְּבָר אֵין כְּלוּם וְלִי יֵשׁ אֶת הֶעָתִיד
אֲנִי שָׂם אוֹתוֹ עָמֹק בַּכִּיס וְלֹא מִסְתַּכֵּל
חַי עַל טַיָּס אוֹטוֹמָטִי
מְרַחֵף כְּמוֹ דָּג בַּמַּיִם
עוֹבֵר אֹרַח בַּסְּעָרָה
אֵין אֲרוּחוֹת חִנָּם וְאֵין הִזְדַּמְּנוּת שְׁנִיָּה
וּבְסוֹפוֹ שֶׁל דָּבָר זֶה דָּבָר שֶׁל שִׁגְרָה
